Basic syntax

    match some_value:

        case pattern1:
            # ... code ...
        
        case pattern2:
            # ... code ...
        
        # ... etc

New: soft keywords

Matching a sequence

    command = "go north"

    match command.split():  # ['go', 'north']

        case [action, obj]:
            # action == 'go'
            # obj == 'north'
            ...

Multiple patterns

    match command.split():

        case [action]:
            # interpret single-verb action
            ...

        case [action, obj]:
            # interpret action, obj 
            ...

Specific values

    match command.split():
        
        case ["quit"]:
            quit_game()

        case ["look"]:
            room.describe()

        case ["get", obj]:
            character.get(obj, room)

        case ["go", direction]:
            room = room.neighbor(direction)

Default case

    match command.split():
        
        case ["quit"]: ...

        case ["look"]: ...

        case ["get", obj]: ...

        case ["go", direction]: ...

        case _:
            print(f"I don't understand {command!r}")

Alternatives

    match command.split():

        case ["north"] | ["go", "north"]:
            room = room.neighbor("north")

        case ["get", o] | ["pick", "up", o] | ["pick", o, "up"]:
            ... # Code for picking up object o

Capturing sub-patterns

    match command.split():

        case ["go", ("north" | "south" | "east" | "west")]:
            # how do I know which direction to go?
    
    match command.split():

        case ["go", ("north" | "south" | "east" | "west") as d]:
            room = room.neighbor(d)

Conditions

    match command.split():

        case ["go", direction] if direction in room.exits:
            room = room.neighbor(direction)

        case ["go", _]:
            print("Sorry, you can't go that way")

Matching mappings

    match action:

        case {"text": message, "color": c}:
            ui.set_text_color(c)
            ui.display(message)

        case {"sleep": duration}:
            ui.wait(duration)

        case {"sound": url, "format": "ogg"}:
            ui.play(url)

        case {"sound": _, "format": _}:
            warning("Unsupported audio format")

Matching via classes

    match event:

        case Click(position=(x, y)):
            handle_click_at(x, y)

        case KeyPress("Q") | Quit():
            game.quit()

        case KeyPress("up arrow"):
            game.go_north()

        case KeyPress():
            pass # Ignore other keystrokes

        case other_event:
            raise ValueError(f"Unrecognized: {other_event}")

Intricate

    match points:
        case []:
            print("No points")
        case [Point(0, 0)]:
            print("The origin")
        case [Point(x, y)]:
            print(f"Single point {x}, {y}")
        case [Point(0, y1), Point(0, y2)]:
            print(f"Two on the Y axis at {y1}, {y2}")
        case [Point(x, y)] if x == y:
            print("One on the diagonal")
        case _:
            print("Something else")

Real-world: GitHub hook

    match github_event:

        case {
                "issue": {"closed_at": closed},
                "comment": {"created_at": commented},
            } if closed == commented:
            # A "Close with comment" comment.
            pass

        case {"sender": {"login": who}} if who == my_name:
            # A comment from ourselves.
            pass

        case {"issue": {"pull_request": _}}:
            # A pull request comment.
            return process_pull_request_comment(event)
